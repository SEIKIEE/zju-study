%{
#include <stdio.h>
int wordCount = 0;
int lineCount = 0;
int charCount = 0;
%}
digit [0-9]
alpha [A-Za-z]
line \n
character .
%%
({alpha}|{digit})+ {
	wordCount++;
	charCount += yyleng;
}
{line} {
	lineCount++;
	charCount++;
}
{character} {
	charCount++;
}
%%
int main()
{
	yylex();
	printf("Total character number = %d\n", charCount);
	printf("Total word number = %d\n", wordCount);
	printf("Total line number = %d\n", lineCount);
	return 0;
}
